private void Use_Example_SetGameplayDefinitions()
{
var gameplayDefinitions = new GameplayDefinitions();
gameplayDefinitions.AddDefinition("PlayerClass", (int)PlayerClass.Assault);
gameplayDefinitions.AddDefinition("SceneName", "ExampleLevel");
gameplayDefinitions.AddDefinition("GameMode", (int)GameMode.TDM);
gameplayDefinitions.AddDefinition("RoundStyle", (int)RoundStyle.Rounds);
gameplayDefinitions.AddDefinition("MapName", "Guanaco Desert");
gameplayDefinitions.AddDefinition("Goal", 1000);
gameplayDefinitions.AddDefinition("FriendlyFire", true);
gameplayDefinitions.AddDefinition("MaxPlayers", 5);
int i = 0;
foreach (var playerData in LudeoGameManager.RemotePlayersData)
{
var pos = playerData.V3Position;
gameplayDefinitions.AddDefinition("RemotePlayerName_" + i, "BOT_" + (i + 1));
gameplayDefinitions.AddDefinition("RemotePlayerTeam_" + i, 2);
i++;
}
gameplayDefinitions.AddDefinition("LocalPlayerTeam", 1);
gameplayDefinitions.AddDefinition("TotalRemotePlayers", 6);
var ok = LudeoManager.SetGameplayDefinitions(gameplayDefinitions);
if (ok == LudeoMessage.Success)
ok = LudeoManager.ReadyForGameplay();
Debug.Log("[Response] - SetGameplayDefinitions = " + ok);
}
